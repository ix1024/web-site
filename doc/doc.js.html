/** 
 
 * @description 获取URL参数 
 * @argument {String} 参数名 返回字符串
 * @return {String} result String
 
 
 * @since 2016-08-17
 * @author kingwell
 * @version 1.0.0 
 */
function getQueryValue(name) {
	var reg = new RegExp("(^|&)" + name + "=([^&]*)(&|$)", "i");
	var r = location.search.substr(1).match(reg);
	if (r !== null) {
		return unescape(r[2]);
	}
	return null;
}
